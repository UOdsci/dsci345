kpart <- function(df,k){
  n = dim(df)[1]
  ri <- sample(1:n)
  sp <- c(0,round(quantile(1:n,(1:(k-1))/k)),n)
  
  kfolds <- vector(mode='list', length=k)
  for(i in 1:k){
    ind_test <- ri[(sp[i]+1):sp[(i+1)]]
    ind_train <- (1:n)[-ind_test]
    test_df <- df[ind_test,]
    train_df <- df[ind_train,]
    kfolds[[i]] <- list(train=train_df,test=test_df)
  }
  return(kfolds)
}
xyk <- kpart(xy,5)

rmsepd <- 1:10
